# 单词出现次数

```c++
#include<iostream>
#include<cctype>
#include<string>
using namespace std;

int main()
{
	string a,b;
	getline(cin,a);
    getline(cin,b);
	int len1=a.length();
	int len2=b.length();
	for(int i=0;i<len1;i++)
	{
		
		a[i]=tolower(a[i]);
	}	
	for(int i=0;i<len2;i++)
	{
		b[i]=tolower(b[i]);
	}
	a=' '+a+' ';
	b=' '+b+' ';
	if(b.find(a)==string::npos)
	{
		cout<<-1<<endl;
	}
	else
	{
		int x=b.find(a);
		int y=b.find(a),s=0;
		while(y!=string::npos)
		{
			++s;
			y=b.find(a,y+1);
		}
		cout<<s<<' '<<x<<endl;
	}
	
} 
```
